# ------------------------------- 
# Log Levels	
# ------------------------------- 
SET(REX_LOG_LEVEL_OFF 0)
SET(REX_LOG_LEVEL_FATAL 1)
SET(REX_LOG_LEVEL_ERROR 2)
SET(REX_LOG_LEVEL_WARN 3)
SET(REX_LOG_LEVEL_INFO 4)
SET(REX_LOG_LEVEL_DEBUG 5)
SET(REX_LOG_LEVEL_TRACE 6)

SET(DEBUG_OFF Off)
SET(DEBUG_FATAL Fatal)
SET(DEBUG_ERROR Error)
SET(DEBUG_WARN Warning)
SET(DEBUG_INFO Info)
SET(DEBUG_DEBUG Debug)
SET(DEBUG_TRACE Trace)

SET(LOG_LEVELS ${DEBUG_OFF} ${DEBUG_FATAL} ${DEBUG_ERROR} ${DEBUG_WARN} ${DEBUG_INFO} ${DEBUG_DEBUG} ${DEBUG_TRACE})
SET(REX_LOG_LEVEL ${DEBUG_DEBUG} CACHE STRING "Setup log level")
